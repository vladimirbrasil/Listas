  function ResizeImage(img) {
    var width = img.clientWidth;
    var height = img.clientHeight;
    var wf = 113 / width;
    var hf = 86 / height;
    if (hf < wf) {
      img.style.width = Math.floor(width * hf) + "px";
      img.style.height = Math.floor(height * hf) + "px";
    } else {
      img.style.width = Math.floor(width * wf) + "px";
      img.style.height = Math.floor(height * wf) + "px";
    };
  };

function $() {
  var elements = new Array();
  for (var i = 0; i < arguments.length; i++) {
    var element = arguments[i];

	function validaFormEndereco(form){
		if (form.endereco.value==''){
			alert('Preencha o campo logradouro.');
			form.endereco.focus();
			return false;
		}
		
		if (form.numero.value != ''){
			if (!form.numero.value.match(/\d+/)){
				alert('Numero incorreto.');
				form.numero.select();
				return false;
			}
		}
		
		if (form.uf_busca.selectedIndex==0){
			alert('Selecione o estado.');
			form.uf_busca.focus();
			return false;
		}
		
		if (form.cod_localidade.value == '0' || form.cod_localidade.value == ''){
			alert('Selecione a cidade.');
			return false;
		}
	}

	function validaFormTelefone(form){
		if (form.telefone.value==''){
			alert('Preencha o campo telefone.');
			form.telefone.focus();
			return false;
		}
		else{
			if (form.telefone.value.match(/^(0800|\d{3,4})\-*\d\d\d\d\d$/) || form.telefone.value.match(/^0(3|8)00\-*\d{6,9}$/)){
				if (form.telefone.value.match(/^0800/)){
					form.ddd.value = '';
				}
			}
			else{
				alert ('Telefone incorreto.');
				form.telefone.select();
				return false;
			}
		}
		return true;
	}
